<script>
  const slider = document.getElementById('slider');
  let currentSlide = 0;
  const totalSlides = 5;

  function updateSlide() {
    slider.style.transform = `translateX(-${currentSlide * 100}vw)`;
  }

  // Global tap navigation (disabled for interactive words)
  document.addEventListener('click', (e) => {
    if (e.target.closest('.interactive')) return; // don't slide on interactive taps
    const mid = window.innerWidth / 2;
    if (e.clientX > mid && currentSlide < totalSlides - 1) currentSlide++;
    if (e.clientX < mid && currentSlide > 0) currentSlide--;
    updateSlide();
  });

  // Swipe navigation
  let touchStartX = 0;
  document.addEventListener('touchstart', (e) => {
    touchStartX = e.touches[0].clientX;
  });

  document.addEventListener('touchend', (e) => {
    if (e.target.closest('.interactive')) return; // ignore interactive
    const endX = e.changedTouches[0].clientX;
    const diff = touchStartX - endX;
    if (diff > 50 && currentSlide < totalSlides - 1) currentSlide++;
    if (diff < -50 && currentSlide > 0) currentSlide--;
    updateSlide();
  });

  function haptic() {
    if (navigator.vibrate) navigator.vibrate([20, 30, 20]);
  }

  function handleInteractive(event, emojis) {
    event.stopPropagation();
    haptic();
    burstAllAround(emojis);
  }

  function burstAllAround(emojis) {
    const count = 26; // explosive but tasteful
    for (let i = 0; i < count; i++) {
      const span = document.createElement('span');
      span.className = 'emoji';
      span.textContent = emojis[Math.floor(Math.random() * emojis.length)];
      span.style.left = Math.random() * window.innerWidth + 'px';
      span.style.top = Math.random() * window.innerHeight + 'px';
      document.body.appendChild(span);
      setTimeout(() => span.remove(), 1600);
    }
  }
</script>
